63 cubic_bezier_poly_coeff(iterator b, Point *pc) {
64  double c[10] = {1,
65  -3, 3,
66  3, -6, 3,
67  -1, 3, -3, 1};
68 
69  int cp = 0;
70 
71  for(int i = 0; i < 4; i++) {
72  pc[i] = Point(0,0);
73  ++b;
74  }
75  for(int i = 0; i < 4; i++) {
76  --b;
77  for(int j = 0; j <= i; j++) {
78  pc[3 - j] += c[cp]*(*b);
79  cp++;
80  }
81  }
82 }
